/*
Express JS Boilerplate

Initialize a Node js app:
npm init -y

Install Typescript:
npm install typescript -D
npm install @types/node -D

Initialize tsconfig.json:
npx tsc --init

Modify tsconfig.json:
Uncomment "outDir" and set it to "./build"
Uncomment "removeComments": true

Modify package.json and overwrite:
"scripts": {
  "build": "tsc",
  "start": "tsc && nodemon ./build/index"
}

Install express:
npm install express
npm install @types/express -D

Install nodemon:
npm install nodemon -D

Install dotenv:
npm install dotenv

Create a .env file in the root of your project and enter the following.
PORT=3000
*/
